```typescript
import { useAlarm } from './hooks/useAlarm';
import Dashboard from './components/Dashboard';

export default function App() {

  const showCountdown = alarm.isRunning || alarm.phase !== 'idle';

  return (
    <div className="min-h-dvh bg-bg text-text-primary flex flex-col items-center">
      {showCountdown ? (

          <p className="text-2xl text-text-secondary">Alarm Active</p>
          <p className="mt-2">Phase: {alarm.phase}</p>
          <button onClick={alarm.stop} className="mt-4 px-6 py-3 bg-accent text-white rounded-xl">
            Stop
          </button>
        </div>
      ) : (
        <Dashboard alarm={alarm} />
      )}
    </div>
  );
}
```

| playTestSound() → AudioContext | Requires prior user gesture; wrapped in try/catch |
| sessionStorage → IosInstallBanner | Stores only a boolean flag — no sensitive data |
